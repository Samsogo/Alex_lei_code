git上传文件步骤
	1.git add .
	2.git commit -m '注释'
	3.git push -u origin master
	若长传不成功可以先同步下：
			git pull --rebase origin master
			git pull origin master
	
git删除文件步骤
	1.同步本地仓库和远程仓库 git pull
	2.删除文件 git rm '文件名'
	3.git commit -m '任意'
	4.git push
	
git删除提交记录步骤
	1.git log 查看日志
	2.git reset --hard （commit去掉）6510c6680d794c8b309e2c59b0e5e5e380866f44 
	3.git push origin HEAD --force

git分支相关命令
	1.新建一个分支slave 	git branch slave
	2.查看当前分支		git branch
	3.切换分支		git checkout slave
	4.创建并切换分支slave1	git checkout -b slave1
	5.合并分支(当前的与要合并的)		git merge slave
	合并之前当前分支看不到其他分支的文件，合并之后可以看到合并过的分支的文件，需要git push下
	6.删除分支		git branch -d branch_name
	7.恢复分支		git branch branch_name HEAD{4}
